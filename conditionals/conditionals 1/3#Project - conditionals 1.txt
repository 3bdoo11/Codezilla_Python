#list of countries and their weekends
countries_fri_sat = ['egypt', 'saudi arabia', 'ksa', 
'kuwait'] 
countries_sat_sun = ['australia', 'usa', 'united states', 
'united kingdom', 'uk']
#ask about country name
country = input("Enter country name: ").strip().lower()

if country in countries_fri_sat:
    print(f"Friday and Saturday are the weekends in {country.upper()}")
elif country in countries_sat_sun:
    print(f"Saturday and Sunday are the weekends in {country.upper()}")
else:
    print("Sorry,I don't know")
