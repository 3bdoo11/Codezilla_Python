first_num = float(input("Enter The First Number: "))
operator = input('Enter the Operator: ')
second_num = float(input("Enter The Second Number: "))

if operator == "*":
    operation = first_num * second_num
    print(f"Multiplication result is {operation}")
elif operator == "+":
    operation = first_num + second_num
    print(f"collection result is {operation}")
elif operator == "/":
    operation = first_num / second_num
    print(f"divide result is {operation}")
elif operator == "-":
    operation = first_num - second_num 
    print(f"Subtract result is {operation}")
elif operator == "**":
    operation = first_num ** second_num
    print(f"power result is {operation}")
else:
    print("Sorrt, Please Enter Valid Inputs")
